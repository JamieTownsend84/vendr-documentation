// Compare version numbers to see if they are a range

const semver = require("semver")

let prevVersion = "3.2.9"
let nextVersion = "4.0.1"
let diff = semver.diff(prevVersion, nextVersion)
let prevNextVersion = semver.inc(prevVersion, diff)
let isRange = semver.neq(prevNextVersion, nextVersion)

console.log(isRange)
